    s.name AS category_name

CREATE VIEW product_details AS
SELECT
    p.id AS product_id,
    p.product_name,
    b.name AS brand_name,
    c.name AS category_name,
    p.price
FROM products p
JOIN brands b
    ON p.brand_id = b.id
JOIN categories c
    ON p.category_id = c.id;


select * from product_details where price > 1000;
 select * from product_details where brand_name="Dell" and category_name="laptop";
 select * from product_details where category_name="smart phone" and price between 400 and 1500;

select * from product_details order by product_id asc;
